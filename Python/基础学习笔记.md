    	print("Happy Birthday,dear lily!")

    	print("Happy Birthday to you!")

        happy()

        happy()
        
    def main():
        sing('mike')
        print()
        sing('lily')
        print()
        sing('lucy')
    
    main()
    # 为mike、lily、lucy三个人唱生日歌
    ```

  - python蟒蛇绘制实例

    ```
    # PythonDraw.py
    import turtle
    turtle.setup(650,350,200,200)
    turtle.penup()
    turtle.fd(-250)
    turtle.pendown()
    turtle.pensize(25)
    turtle.pencolor("purple")
    turtle.seth(-40)
    for i in range(4):
        turtle.circle(40,80)
        turtle.circle(-40.80)
    turtle.circle(40,80/2)
    turtle.fd(40)
    turtle.circle(16,180)
    turtle.fd(40 * 2/3)
    turtle.done()
